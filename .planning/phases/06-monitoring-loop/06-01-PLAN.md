Create entry point that:
1. Configure logging (INFO level to console, include timestamp and logger name)
2. Load Settings from environment (LCB_ prefix)
3. Initialize and run MonitoringLoop
4. Handle KeyboardInterrupt for clean shutdown

```python
import asyncio
import logging
import sys

from live_coverage_bot.config import Settings
from live_coverage_bot.core import MonitoringLoop

def main() -> int:
    logging.basicConfig(
        level=logging.INFO,
        format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
    )
    logger = logging.getLogger(__name__)

    try:
        settings = Settings()
    except Exception as e:
        logger.error("Failed to load settings: %s", e)
        return 1

    logger.info("Starting Live Coverage Bot (polling every %ds)", settings.polling_interval_seconds)

    loop = MonitoringLoop(settings)
    try:
        asyncio.run(loop.run())
    except KeyboardInterrupt:
        logger.info("Shutdown requested, exiting")

    return 0

if __name__ == "__main__":
    sys.exit(main())
```
